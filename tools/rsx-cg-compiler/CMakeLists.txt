# rsx-cg-compiler — Phase 8 Cg → NV40 compiler
#
# Self-contained build: all donor front-end sources are copied into
# tools/rsx-cg-compiler/src/donor/ rather than referenced from a sibling
# clone of vita-cg-compiler.  Sync against upstream is a manual `cp`
# operation when needed.

cmake_minimum_required(VERSION 3.16)
project(rsx-cg-compiler
    LANGUAGES CXX
    DESCRIPTION "PS3 Cg-to-RSX (NV40) shader compiler (Sony binary-compat)")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "" FORCE)
endif()

if(NOT MSVC)
    add_compile_options(
        -Wall -Wextra
        -Wno-unused-parameter
        -Wno-unused-variable
        -Wno-unused-but-set-variable
        -Wno-unused-function
        -Wno-missing-field-initializers
        -Wno-sign-compare
        -Wno-reorder
        -Wno-switch)
endif()

set(SRC_DIR   ${CMAKE_CURRENT_SOURCE_DIR}/src)
set(DONOR_DIR ${SRC_DIR}/donor)

# ----------------------------------------------------------------------------
# Embedded builtin Cg header (generated by donor's embed_builtin_header.py).
# ----------------------------------------------------------------------------

find_package(Python3 REQUIRED COMPONENTS Interpreter)

set(EMBED_INPUT  ${DONOR_DIR}/builtin_shader_header.h)
set(EMBED_OUTPUT ${DONOR_DIR}/builtin_shader_header_embedded.inc)
set(EMBED_SCRIPT ${DONOR_DIR}/embed_builtin_header.py)

add_custom_command(
    OUTPUT  ${EMBED_OUTPUT}
    COMMAND ${Python3_EXECUTABLE} ${EMBED_SCRIPT} ${EMBED_INPUT} ${EMBED_OUTPUT}
    DEPENDS ${EMBED_INPUT} ${EMBED_SCRIPT}
    COMMENT "Embedding builtin shader header (${EMBED_OUTPUT})"
    VERBATIM)

# ----------------------------------------------------------------------------
# Source lists.
# ----------------------------------------------------------------------------

# Donor front-end sources.
set(SRC_FRONTEND_DONOR
    ${DONOR_DIR}/common/errorHandler.cpp
    ${DONOR_DIR}/common/memoryPool.cpp
    ${DONOR_DIR}/frontend/preprocessor.cpp
    ${DONOR_DIR}/frontend/lexer.cpp
    ${DONOR_DIR}/frontend/parser.cpp
    ${DONOR_DIR}/frontend/semantic.cpp
    ${DONOR_DIR}/frontend/symbol_table.cpp
    ${DONOR_DIR}/frontend/types.cpp
    ${DONOR_DIR}/builtin_shader_header.cpp
)

# Donor IR layer — hardware-independent IRModule built from AST.
# The donor's USSE back-end (src/backend/, src/gxp/) is not brought in:
# NV40 lowering lives in the SRC_NV40 list below.
set(SRC_IR_DONOR
    ${DONOR_DIR}/ir/ir.cpp
    ${DONOR_DIR}/ir/ir_builder.cpp
    ${DONOR_DIR}/ir/ir_passes.cpp
)

# NV40 back-end (this tree).  Register-file / opcode headers are adopted
# from Mesa/Nouveau via PSL1GHT cgcomp; per-stage lowering is authored here.
set(NV40_DIR ${SRC_DIR}/nv40)
set(SRC_NV40
    ${NV40_DIR}/nv40_emit.cpp
    ${NV40_DIR}/nv40_vp_assembler.cpp
    ${NV40_DIR}/nv40_vp_emit.cpp
    ${NV40_DIR}/nv40_fp_emit.cpp
)

# NEW: Sony container emitter (this tree).  Populated in Stage 4.
set(SRC_CONTAINER
    # ${SRC_DIR}/sony_container_fp.cpp
    # ${SRC_DIR}/sony_container_vp.cpp
)

# Driver / entry point.
set(SRC_MAIN
    ${SRC_DIR}/main.cpp
)

add_executable(rsx-cg-compiler
    ${SRC_FRONTEND_DONOR}
    ${SRC_IR_DONOR}
    ${SRC_NV40}
    ${SRC_CONTAINER}
    ${SRC_MAIN}
    ${EMBED_OUTPUT}
)

target_include_directories(rsx-cg-compiler PRIVATE
    ${DONOR_DIR}
    ${DONOR_DIR}/common
    ${DONOR_DIR}/frontend
    ${DONOR_DIR}/ir
    ${SRC_DIR}
    ${NV40_DIR}
)
