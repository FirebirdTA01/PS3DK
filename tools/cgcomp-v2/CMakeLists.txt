# cgcomp-v2 — Phase 8 Cg → NV40 compiler
#
# Self-contained build: all donor front-end sources are copied into
# tools/cgcomp-v2/src/donor/ rather than referenced from a sibling
# clone of vita-cg-compiler.  Sync against upstream is a manual `cp`
# operation when needed.

cmake_minimum_required(VERSION 3.16)
project(cgcomp-v2
    LANGUAGES CXX
    DESCRIPTION "PS3 Cg-to-NV40 shader compiler (Sony binary-compat)")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

if(NOT CMAKE_BUILD_TYPE AND NOT CMAKE_CONFIGURATION_TYPES)
    set(CMAKE_BUILD_TYPE Release CACHE STRING "" FORCE)
endif()

if(NOT MSVC)
    add_compile_options(
        -Wall -Wextra
        -Wno-unused-parameter
        -Wno-unused-variable
        -Wno-unused-but-set-variable
        -Wno-unused-function
        -Wno-missing-field-initializers
        -Wno-sign-compare
        -Wno-reorder
        -Wno-switch)
endif()

set(SRC_DIR   ${CMAKE_CURRENT_SOURCE_DIR}/src)
set(DONOR_DIR ${SRC_DIR}/donor)

# ----------------------------------------------------------------------------
# Embedded builtin Cg header (generated by donor's embed_builtin_header.py).
# ----------------------------------------------------------------------------

find_package(Python3 REQUIRED COMPONENTS Interpreter)

set(EMBED_INPUT  ${DONOR_DIR}/builtin_shader_header.h)
set(EMBED_OUTPUT ${DONOR_DIR}/builtin_shader_header_embedded.inc)
set(EMBED_SCRIPT ${DONOR_DIR}/embed_builtin_header.py)

add_custom_command(
    OUTPUT  ${EMBED_OUTPUT}
    COMMAND ${Python3_EXECUTABLE} ${EMBED_SCRIPT} ${EMBED_INPUT} ${EMBED_OUTPUT}
    DEPENDS ${EMBED_INPUT} ${EMBED_SCRIPT}
    COMMENT "Embedding builtin shader header (${EMBED_OUTPUT})"
    VERBATIM)

# ----------------------------------------------------------------------------
# Source lists.
# ----------------------------------------------------------------------------

# Donor front-end sources.  Stage 1 ships these; Stage 2 may drop / replace
# pieces (e.g. semantic analysis if it depends on USSE-specific bits).
set(SRC_FRONTEND_DONOR
    ${DONOR_DIR}/common/errorHandler.cpp
    ${DONOR_DIR}/common/memoryPool.cpp
    ${DONOR_DIR}/frontend/preprocessor.cpp
    ${DONOR_DIR}/frontend/lexer.cpp
    ${DONOR_DIR}/frontend/parser.cpp
    ${DONOR_DIR}/frontend/symbol_table.cpp
    ${DONOR_DIR}/frontend/types.cpp
    ${DONOR_DIR}/builtin_shader_header.cpp
    # Stage 2:
    #   semantic.cpp  — needs RSX-aware shader-stage hooks
    #   ir/*.cpp      — IR builder + passes once we wire NV40 lowering
)

# NEW: NV40 back-end (this tree).  Populated in Stage 2.
set(SRC_NV40
    # ${SRC_DIR}/nv40_lowering.cpp
    # ${SRC_DIR}/nv40_regalloc.cpp
    # ${SRC_DIR}/nv40_emit.cpp
)

# NEW: Sony container emitter (this tree).  Populated in Stage 4.
set(SRC_CONTAINER
    # ${SRC_DIR}/sony_container_fp.cpp
    # ${SRC_DIR}/sony_container_vp.cpp
)

# Driver / entry point.
set(SRC_MAIN
    ${SRC_DIR}/main.cpp
)

add_executable(cgcomp-v2
    ${SRC_FRONTEND_DONOR}
    ${SRC_NV40}
    ${SRC_CONTAINER}
    ${SRC_MAIN}
    ${EMBED_OUTPUT}
)

target_include_directories(cgcomp-v2 PRIVATE
    ${DONOR_DIR}
    ${DONOR_DIR}/common
    ${DONOR_DIR}/frontend
    ${SRC_DIR}
)
